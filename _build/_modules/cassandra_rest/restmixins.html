from cassandraserializer.serializers import CQLManager
from flask import request
from decorators import api_method
from webargs.flaskparser import FlaskParser
from flask.views import MethodView

parser = FlaskParser()


class APIBase(MethodView):
    serializer = None
    model = None
    fields = None
    preprocessors = None
    postprocessors = None
    pks = None

    def __init__(self, model, fields, preprocessors=None, postprocessors=None, pks=None, ignore_pks_on_update=True):
        self.model = model
        self.fields = fields
        self.preprocessors = preprocessors
        self.postprocessors = postprocessors
        self.pks = pks
        self.ignore_pks_on_update = ignore_pks_on_update

    def queryset(self):
        return self.model.objects.all()

    def _get_manager(self):
        if self.serializer is None:
            self.serializer = CQLManager(self.model, self.fields, viewset=self,
                                         ignore_pks_on_update=self.ignore_pks_on_update)
        return self.serializer


class Create(APIBase):
    @api_method
    def create(self):
        serializer = self._get_manager()
        kwargs = parser.parse(serializer.arg_parser, request, targets=('form', 'json', 'url_params'))
        return serializer.create(**kwargs)


class RetrieveList(APIBase):
    @api_method
    def retrieve_list(self, **filters):
        serializer = self._get_manager()
        return serializer.retrieve_list(**filters)


class RetrieveSingle(APIBase):
    @api_method
    def retrieve(self, **primary_keys):
        serializer = self._get_manager()
        return serializer.retrieve(primary_keys)


class Update(APIBase):
    @api_method
    def update(self, **primary_keys):
        serializer = self._get_manager()
        kwargs = parser.parse(serializer.arg_parser, request, targets=('form', 'json', 'url_params'))
        return serializer.update(primary_keys, **kwargs)


class Delete(APIBase):
    @api_method
    def remove(self, **primary_keys):
        serializer = self._get_manager()
        return serializer.delete(primary_keys)
