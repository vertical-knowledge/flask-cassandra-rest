class CreateRetrieveDeleteViewSet(Create, RetrieveSingle, RetrieveList, Delete):
    def get(self, **kwargs):
        for f in self.fields:
            if f not in kwargs:
                return self.retrieve_list()
        return self.retrieve()

    def post(self, **kwargs):
        return self.create()

    def delete(self, **kwargs):
        return self.remove(**kwargs)

[docs]def register_viewset(app, viewset, model, fields=None, pks=None,
                     endpoint=None, url=None, preprocessors=None, postprocessors=None, ignore_pks_on_update=True):
    """
    Sets up the viewset to respond to url endpoints for the specified model

    :param app: The flask app to use
    :type app: flask.Flask
    :param viewset: The cassandra_rest viewset to use
    :type viewset: flask.views.View
    :param model: The cqlengine model to use
    :type model: cqlengine.models.Model
    :param fields: The fields on the model to use.  By default it will grab all of the fields on the specified model
    :type fields: list
    :param pks: The primary keys on which the API will respond to GET, PUT, and DELETE operations. The url for these operations will be <url>/<pk1>/<pk2>/.../<pkn>
    :type pks: list
    :param endpoint: The name of the endpoint.  Defaults to <model>_api
    :type endpoint: str
    :param url: The base url on which the api for this model responds
    :type url: str
    :return: None
    """
    model_name = convert_to_underscore(model.__name__)
    if url is None:
        url = '/{0}'.format(model_name)
    if endpoint is None:
        endpoint = '{0}_api'.format(model_name)
    if fields is None:
        raise ValueError('The kwarg "fields" is required')
    if type(fields) is not list:
        raise TypeError('The "fields" kwarg must be a list')
    if pks is None:
        raise ValueError('The kwarg "pks" is required')
    if type(pks) is not list:
        raise TypeError('pks kwarg must be of type \'list\'')

    view_func = viewset.as_view(endpoint, model, fields,
                                preprocessors=preprocessors, postprocessors=postprocessors,
                                pks=list(pks), ignore_pks_on_update=ignore_pks_on_update)
    app.add_url_rule(url, view_func=view_func, methods=['GET', 'POST'])
    if pks is not None and len(pks) > 0:
        for i in range(len(pks)):
            pks[i] = '<{0}>'.format(pks[i])
        pks_encoded = '/'.join(pks)
        app.add_url_rule('{0}/{1}'.format(url, pks_encoded), view_func=view_func, methods=['GET', 'PUT', 'DELETE'])



_first_cap_re = re.compile('(.)([A-Z][a-z]+)')
_all_cap_re = re.compile('([a-z0-9])([A-Z])')


def convert_to_underscore(toconvert):
    """
    Converts a string from CamelCase to underscore

    :param toconvert: The string to convert from CamelCase to underscore (i.e. camel_case)
    :type toconvert: str
    :return: The converted string
    :rtype: str
    """
    i = 0
    while toconvert[i] == '_':
        i += 1
    prefix = None
    if i > 0:
        prefix = toconvert[0:i]
        toconvert = toconvert[i:]
    s1 = _first_cap_re.sub(r'\1_\2', toconvert)
    s2 = _all_cap_re.sub(r'\1_\2', s1).lower()
    if prefix is not None:
        return '{0}{1}'.format(prefix, s2)
    return s2
